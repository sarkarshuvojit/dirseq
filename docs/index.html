Common Usage Patterns
# Experiment directories
mkdir experiment-$(dirseq)
# Versioned outputs
./build.sh > build-$(dirseq).log
# Padded sequences
touch session-$(dirseq -p 3).txt
# Custom starting point
dirset set-seq 100
dirseq # outputs: 101
